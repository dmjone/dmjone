Problem Statement

Unexpected Demand

A widget manufacturer is facing unexpectedly high demand for its new product,. They would like to satisfy as many customers as possible. Given a number of widgets available and a list of customer orders, what is the maximum number of orders the manufacturer can fulfill in full?

Complete the function filledOrders in the editor below. The function must return a single integer denoting the maximum possible number of fullfilled orders.

filledOrders has the folowing parameter(s):
  order: an array of integers listing the orders

Constraints
  • $1\leq n \leq 2\times10^5$
  • $1\leq order[i] \leq 10^9$
  • $1\leq k \leq 10^9$

int filledOrders(vector order, int k) {
    sort(order.begin(), order.end());           // sort the order sizes in non-decreasing order
    int numFilledOrders = 0;                    // initialize the count of filled orders to zero
    for (int i = 0; i < order.size(); i++) {    // iterate through the sorted order sizes
        if (order[i] <= k) {                    // if the current order can be filled with the available inventory
            numFilledOrders++;                  // increment the count of filled orders
            k -= order[i];                      // reduce the available inventory by the size of the filled order
        } else {                                // if the current order cannot be filled with the available inventory
            break;                              // exit the loop early since no more orders can be filled
        }
    }
    return numFilledOrders;                     // return the count of filled orders
}

int main()
{
    ofstream fout(getenv("OUTPUT_PATH"));

    string order_count_temp;

    int order_count = stoi(ltrim(rtrim(order_count_temp)));

    vector order(order_count);

    for (int i = 0; i < order_count; i++) {
        string order_item_temp;
        getline(cin, order_item_temp);

        int order_item = stoi(ltrim(rtrim(order_item_temp)));

        order[i] = order_item;
    }

    string k_temp;
    getline(cin, k_temp);

    int k = stoi(ltrim(rtrim(k_temp)));

    int result = filledOrders(order, k);

    fout << result << "\n";

    fout.close();

    return 0;
}

string ltrim(const string &str) {
    string s(str);

    s.erase(
        s.begin(),
        find_if(s.begin(), s.end(), not1(ptr_fun(isspace)))
    );

    return s;
}

string rtrim(const string &str) {
    string s(str);

    s.erase(
        find_if(s.rbegin(), s.rend(), not1(ptr_fun(isspace))).base(),
        s.end()
    );

    return s;
}
