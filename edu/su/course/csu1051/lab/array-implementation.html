Objective

To understand the implementation an array in C++

#include <iostream>
int main() {
    // Declare an array of size 5
    int arr[5];

    // Assign values to the array
    arr[0] = 1;
    arr[1] = 2;
    arr[2] = 3;
    arr[3] = 4;
    arr[4] = 5;

    // Print the values of the array
    for (int i = 0; i < 5; i++) {
        std::cout << "Element at " << i << " is " << arr[i] << std::endl;
    }

    return 0;
}
